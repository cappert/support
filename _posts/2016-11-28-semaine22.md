```javascript
let Bavard = (superclass) => class extends superclass {
  cause(){
    console.log('blabla');
  }
}

class Muet{
  causePas(){
    console.log('...');
  }
}

class MuetBavard extends Bavard(Muet){
  constructor(){
    super();
    super.causePas();
  }
} 

let mb = new MuetBavard();
mb.cause();
```
